### الفرق الأساسي بين ووردبريس و Next.js في الـ SEO

- **ووردبريس:** يولد HTML جاهزًا من الخادم (Server-Side)
- **React (افتراضيًا):** يعتمد على JavaScript في المتصفح (Client-Side Rendering) مما قد يسبب مشاكل لمحركات البحث
- **Next.js:** يحل هذه المشكلة بـ **Server-Side Rendering (SSR)** و **Static Site Generation (SSG)**

---

### كيفية تطبيق معايير الـ SEO على موقعك

#### 1. **إعداد Next.js لتحسين الـ SEO**

**في `next.config.js`:**
```javascript
/** @type {import('next').NextConfig} */
const nextConfig = {
  reactStrictMode: true,
  images: {
    domains: ['your-domain.com'], // لتحسين الصور
  },
  // إعدادات مهمة للـ SEO
  async headers() {
    return [
      {
        source: '/(.*)',
        headers: [
          {
            key: 'X-Content-Type-Options',
            value: 'nosniff'
          }
        ],
      },
    ]
  }
}

module.exports = nextConfig
```

#### 2. **إنشاء نظام للـ Meta Tags (كبديل لإضافة Yoast SEO)**

**إنشاء ملف `components/Seo.jsx`:**
```jsx
import Head from 'next/head'

const Seo = ({ 
  title, 
  description, 
  keywords, 
  image, 
  url, 
  canonical 
}) => {
  return (
    <Head>
      {/* الأساسيات */}
      <title>{title}</title>
      <meta name="description" content={description} />
      <meta name="keywords" content={keywords} />
      <link rel="canonical" href={canonical} />
      
      {/* Open Graph (لشبكات التواصل الاجتماعي) */}
      <meta property="og:title" content={title} />
      <meta property="og:description" content={description} />
      <meta property="og:image" content={image} />
      <meta property="og:url" content={url} />
      <meta property="og:type" content="website" />
      
      {/* Twitter Card */}
      <meta name="twitter:card" content="summary_large_image" />
      <meta name="twitter:title" content={title} />
      <meta name="twitter:description" content={description} />
      <meta name="twitter:image" content={image} />
    </Head>
  )
}

export default Seo
```

#### 4. **إنشاء Sitemap (خريطة الموقع)**

**إنشاء `pages/sitemap.xml.js`:**
```jsx
const EXTERNAL_DATA_URL = 'https://your-api.com/articles'

function generateSiteMap(posts) {
  return `<?xml version="1.0" encoding="UTF-8"?>
   <urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
     <url>
       <loc>https://your-site.com</loc>
       <lastmod>2023-10-01</lastmod>
       <changefreq>weekly</changefreq>
       <priority>1.0</priority>
     </url>
     ${posts.map(({ id, updatedAt }) => {
       return `
       <url>
           <loc>${`${EXTERNAL_DATA_URL}/${id}`}</loc>
           <lastmod>${updatedAt}</lastmod>
           <changefreq>monthly</changefreq>
           <priority>0.8</priority>
       </url>
     `}).join('')}
   </urlset>
 `
}

export async function getServerSideProps({ res }) {
  const request = await fetch(EXTERNAL_DATA_URL)
  const posts = await request.json()

  const sitemap = generateSiteMap(posts)

  res.setHeader('Content-Type', 'text/xml')
  res.write(sitemap)
  res.end()

  return {
    props: {},
  }
}

export default function SiteMap() {}
```

#### 5. **إعداد ملف `robots.txt`**

**إنشاء `pages/robots.txt`:**
```jsx
const Robots = () => null

export async function getServerSideProps({ res }) {
  const robots = `User-agent: *
Allow: /
Disallow: /api/
Sitemap: https://your-site.com/sitemap.xml`

  res.setHeader('Content-Type', 'text/plain')
  res.write(robots)
  res.end()

  return {
    props: {},
  }
}

export default Robots
```

### مثال عملي لصفحة مقال

```jsx
// pages/articles/[slug].js
import Seo from '../../components/Seo'
import Image from 'next/image'

export default function Article({ article }) {
  return (
    <>
      <Seo 
        title={article.title}
        description={article.excerpt}
        keywords={article.tags.join(', ')}
        image={article.featuredImage}
        url={`https://your-site.com/articles/${article.slug}`}
        canonical={`https://your-site.com/articles/${article.slug}`}
      />
      
      <article className="container">
        <h1>{article.title}</h1>
        
        <Image
          src={article.featuredImage}
          alt={article.imageAlt}
          width={1200}
          height={630}
          priority
        />
        
        <div dangerouslySetInnerHTML={{ __html: article.content }} />
      </article>
    </>
  )
}

export async function getServerSideProps({ params }) {
  // جلب البيانات من API
  const res = await fetch(`https://your-api.com/articles/${params.slug}`)
  const article = await res.json()

  return {
    props: {
      article,
    },
  }
}
```
